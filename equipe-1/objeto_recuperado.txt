import subprocess
import time


class Analise():
    def __init__(self, algoritimos: list, diretorio: str, num_tests: int = 13):
        self.algoritimos = algoritimos
        self.diretorio = diretorio
        self.num_tests = num_tests
    
    def run_binary(self, binary: str, input_file: str):
        start_time = time.time()
        comando = f"./{binary} {input_file}"
        print(comando)
        process = subprocess.Popen(comando, 
                                   stdout=subprocess.PIPE, 
                                   stderr=subprocess.PIPE, 
                                   universal_newlines=True)
        stdout, stderr = process.communicate()
        print(stdout)
        print(stderr)

        end_time = time.time()
        return end_time - start_time
